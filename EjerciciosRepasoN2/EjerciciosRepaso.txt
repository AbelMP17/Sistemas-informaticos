9. 
a) grep "^aa" ficherotexto.txt // que empieza por aa
b) grep "ba$" ficherotexto.txt // que termina por ba
c) grep -E "^a.*b$" ficherotexto.txt // que empieza por a y termina por b, el . significa cualquier caracter y el * significa de uno a muchos
d) grep "\ba" ficherotexto.txt // lineas con palabras que empiezan por "a", se usa \b para que vaya palabra por palabra
e) grep -v '\ba[a-z]*\b' ficherotexto.txt // lineas con palabras que terminan en cualquier cosa menos en "a" 
f) grep -vE "a{3}" ficherotexto.txt // linea qu no contiene cadenas con 3 "a" consecutivas
g) grep -iE "^a{2}[b-z]" ficherotexto.txt //linea que empieze por 2 a pero el tercer caracter se otro
h) grep -E "(ba){2}" ficherotexto.txt // linea que contiene 2 veces ba
i) grep -v "\bb" ficherotexto.txt // lineas que no contienen palabras con b
j) grep -E "\.{3}" ficherotexto.txt // linea que contiene tres . seguidos
k) grep -E "^a|^h" ficherotexto.txt // lineas que empiezan por a o por h
l) grep -v "a$" ficherotexto.txt // lineas que no terminan por a
